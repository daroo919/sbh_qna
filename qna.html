async function loadAnswers(qId, qData, container) { // 답변을 생성 순서대로 가져옵니다. const snap = await getDocs(query(collection(db, "questions", qId, "answers"), orderBy("createdAt", "asc"))); container.innerHTML = ""; // 중복 방지를 위해 비워줌 for (const aDoc of snap.docs) { const aData = aDoc.data(), aId = aDoc.id, isAcc = qData.acceptedAnswerId === aId; // 답변자의 최신 뱃지 정보 가져오기 const userSnap = await getDoc(doc(db, "users", aData.uid)); const aBadge = (userSnap.exists() && userSnap.data().currentBadge) ? userSnap.data().currentBadge : ""; const isMyAnswer = auth.currentUser.uid === aData.uid; const item = document.createElement("div"); item.className = `answer-item ${isAcc ? 'accepted-answer' : ''}`; // HTML 구성: 뱃지 + 닉네임 + 내용 + [채택버튼] + [삭제버튼] item.innerHTML = ` ↳ ${aBadge} ${aData.nickname}: ${aData.content} ${auth.currentUser.uid === qData.uid && !qData.acceptedAnswerId ? `` : ''} ${isMyAnswer && !isAcc ? `` : ''} `; // 채택 버튼 클릭 이벤트 if (item.querySelector(".accept-btn")) { item.querySelector(".accept-btn").onclick = async () => { if(confirm("이 답변을 채택하시겠습니까? 답변자에게 10P가 지급됩니다.")) { await updateDoc(doc(db, "questions", qId), { acceptedAnswerId: aId }); await updateDoc(doc(db, "users", aData.uid), { points: increment(10) }); // location.reload() 대신 상태 반영을 위해 다시 로드하거나 관찰자 이용 } }; } // 답변 삭제 버튼 클릭 이벤트 if (item.querySelector(".answer-delete")) { item.querySelector(".answer-delete").onclick = async () => { if(confirm("답변을 삭제하시겠습니까?")) { try { // 1. 답변 문서 삭제 await deleteDoc(doc(db, "questions", qId, "answers", aId)); // 2. 질문의 답변 개수 1 차감 await updateDoc(doc(db, "questions", qId), { answerCount: increment(-1) }); alert("삭제되었습니다."); } catch (e) { console.error(e); alert("삭제 중 오류가 발생했습니다."); } } }; } container.appendChild(item); } }
